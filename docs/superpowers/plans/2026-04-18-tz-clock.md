> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a live-updating terminal timezone dashboard with weather, installable as a single command.

**Architecture:** Single Python file split internally into pure-logic functions (testable) and a thin curses UI layer. Weather fetched via `wttr.in` JSON API in background threads. Config loaded from defaults → file → CLI args.

**Tech Stack:** Python 3.9+ stdlib: `curses`, `urllib.request`, `json`, `argparse`, `datetime`, `zoneinfo`, `threading`, `time`


4. **Rendering** — `format_zone_row()`, `format_header()`, `format_footer()`, `pad_row()`
5. **Curses UI** — `draw_dashboard()`, `main_loop()`, `main()`

```python
#!/usr/bin/env python3
"""tz-clock — Terminal Timezone Dashboard"""

import argparse
import json
import os
from collections import namedtuple

Zone = namedtuple("Zone", ["name", "tz", "city", "home"])

DEFAULT_ZONES = [
    Zone("HAWAII", "Pacific/Honolulu", "Hilo", False),
    Zone("WEST COAST", "America/Los_Angeles", "Los Angeles", False),
    Zone("MOUNTAIN", "America/Denver", "Denver", False),
    Zone("CENTRAL", "America/Chicago", "Austin", True),
    Zone("EASTERN", "America/New_York", "New York", False),
    Zone("UTC", "UTC", "London", False),
]

DEFAULT_CONFIG_PATH = os.path.expanduser("~/.config/tz-clock/config.json")
DEFAULT_REFRESH_MINUTES = 30


def load_config(path):
    """Load zones and settings from a JSON config file. Returns (zones, refresh_minutes) or raises FileNotFoundError."""
    with open(path) as f:
        data = json.load(f)
    zones = []
    for z in data.get("zones", []):
        zones.append(Zone(
            name=z["name"],
            tz=z["tz"],
            city=z["city"],
            home=z.get("home", False),
        ))
    refresh = data.get("refresh_minutes", DEFAULT_REFRESH_MINUTES)
    return zones, refresh


def parse_args(argv=None):
    """Parse CLI arguments. Returns argparse.Namespace."""
    parser = argparse.ArgumentParser(description="Terminal Timezone Dashboard")
    parser.add_argument("--config", default=None, help="Path to config file")
    parser.add_argument("--no-weather", action="store_true", help="Disable weather")
    parser.add_argument("--refresh", type=int, default=None, help="Weather refresh interval in minutes")
    return parser.parse_args(argv)


def resolve_config(args):
    """Resolve zones and settings from defaults → config file → CLI args."""
    zones = list(DEFAULT_ZONES)
    refresh = DEFAULT_REFRESH_MINUTES

    config_path = args.config or DEFAULT_CONFIG_PATH
    try:
        zones, refresh = load_config(config_path)
    except (FileNotFoundError, json.JSONDecodeError, KeyError):
        if args.config:
            import sys
            print(f"Warning: could not load config from {args.config}", file=sys.stderr)

    if args.refresh is not None:
        refresh = args.refresh

    return zones, refresh, args.no_weather
```

```python
from datetime import datetime
from zoneinfo import ZoneInfo


def get_zone_time(zone):
    """Get current time info for a zone. Returns dict with hour, minute, time_str, date_str, day_str.
    hour/minute are in 24h for progress bar. time_str is 12h without AM/PM."""
    now = datetime.now(ZoneInfo(zone.tz))
    hour_12 = now.hour % 12 or 12
    return {
        "hour": now.hour,
        "minute": now.minute,
        "time_str": f"{hour_12}:{now.minute:02d}",
        "date_str": now.strftime("%b %d"),
        "day_str": now.strftime("%A"),
    }


def make_progress_bar(hour_24, minute, width=14):
    """Build a progress bar string showing position in the 24h day."""
    total_minutes = hour_24 * 60 + minute
    fraction = total_minutes / 1440
    filled = round(fraction * width)
    return "▓" * filled + "░" * (width - filled)
```

```python
# Night clear emoji — used when weatherCode is 113 and local hour is before 6 or after 18
NIGHT_CLEAR_EMOJI = "🌙"


def parse_weather(json_str):
    """Parse wttr.in JSON response. Returns dict with temp_f, temp_c, condition, emoji or None on failure."""
    try:
        data = json.loads(json_str)
        cc = data["current_condition"][0]
        code = cc["weatherCode"]
        emoji = WEATHER_EMOJI.get(code, "?")
        return {
            "temp_f": cc["temp_F"],
            "temp_c": cc["temp_C"],
            "condition": cc["weatherDesc"][0]["value"],
            "emoji": emoji,
        }
    except (json.JSONDecodeError, KeyError, IndexError):
        return None


def format_weather(weather):
    """Format weather dict into display string like '☀️  79°F/26°C Sunny'. Returns '--' if None."""
    if weather is None:
        return "--"
    return f"{weather['emoji']}  {weather['temp_f']}°F/{weather['temp_c']}°C {weather['condition']}"


def fetch_weather(city):
    """Fetch weather for a city from wttr.in. Returns parsed weather dict or None."""
    try:
        url = f"https://wttr.in/{city}?format=j1"
        req = Request(url, headers={"User-Agent": "curl/7.0"})
        with urlopen(req, timeout=10) as resp:
            return parse_weather(resp.read().decode())
    except Exception:
        return None


def fetch_all_weather(zones, results, lock):
    """Fetch weather for all zones in parallel threads. Updates results dict with lock."""
    def worker(zone):
        weather = fetch_weather(zone.city)
        with lock:
            results[zone.name] = weather

    threads = []
    for zone in zones:
        t = threading.Thread(target=worker, args=(zone,), daemon=True)
        t.start()
        threads.append(t)
    for t in threads:
        t.join(timeout=15)
```

```python
BAR_WIDTH = 14


def pad_row(text, width):
    """Pad or truncate text to exact width."""
    if len(text) > width:
        return text[:width]
    return text + " " * (width - len(text))


def format_zone_row(zone, time_info, weather, inner_width=54):
    """Format a single zone row (content between the ║ walls)."""
    marker = " ►" if zone.home else "  "
    name = pad_row(zone.name, 12)
    time_str = f"{time_info['time_str']:>5}"
    bar = make_progress_bar(time_info["hour"], time_info["minute"], BAR_WIDTH)
    weather_str = format_weather(weather)

    content = f"{marker}{name} {time_str}  {bar}  {weather_str}"
    return pad_row(content, inner_width)


def format_header(day_name, date_str, inner_width=54):
    """Format the header row content."""
    left = "  TIMEZONE DASHBOARD"
    right = f"{day_name}, {date_str}"
    gap = inner_width - len(left) - len(right)
    if gap < 1:
        gap = 1
    content = left + " " * gap + right
    return pad_row(content, inner_width)


def format_footer(weather_age_seconds, inner_width=54):
    """Format the footer row content."""
    left = "  r: refresh weather   q: quit"
    if weather_age_seconds is None:
        right = "weather loading..."
    else:
        minutes_ago = int(weather_age_seconds // 60)
        right = f"weather updated {minutes_ago}m ago"
    gap = inner_width - len(left) - len(right)
    if gap < 1:
        gap = 1
    content = left + " " * gap + right
    return pad_row(content, inner_width)
```

```python
def draw_dashboard(stdscr, zones, weather_data, weather_time, no_weather):
    """Draw the full dashboard. Called on every refresh."""
    stdscr.erase()
    height, width = stdscr.getmaxyx()
    inner_width = width - 2  # space between ║ walls

    if width < 40 or height < len(zones) + 6:
        msg = "Terminal too small — please resize"
        stdscr.addstr(height // 2, max(0, (width - len(msg)) // 2), msg)
        stdscr.refresh()
        return

    frame = curses.color_pair(1)
    title_color = curses.color_pair(2) | curses.A_BOLD
    home_color = curses.color_pair(3)
    zone_color = curses.color_pair(4)
    footer_color = curses.color_pair(5) | curses.A_DIM

    row = 0

    # Top border
    top = TOP_LEFT + HORIZONTAL * inner_width + TOP_RIGHT
    stdscr.addstr(row, 0, top[:width], frame)
    row += 1

    # Header
    home_zone = next((z for z in zones if z.home), zones[0])
    home_time = get_zone_time(home_zone)
    header_content = format_header(home_time["day_str"], home_time["date_str"], inner_width)
    stdscr.addstr(row, 0, VERTICAL, frame)
    stdscr.addstr(row, 1, header_content[:inner_width], title_color)
    try:
        stdscr.addstr(row, 1 + inner_width, VERTICAL, frame)
    except curses.error:
        pass
    row += 1

    # Header separator
    sep = T_LEFT + HORIZONTAL * inner_width + T_RIGHT
    stdscr.addstr(row, 0, sep[:width], frame)
    row += 1

    # Zone rows
    for zone in zones:
        time_info = get_zone_time(zone)
        weather = weather_data.get(zone.name) if not no_weather else None
        content = format_zone_row(zone, time_info, weather, inner_width)
        color = home_color if zone.home else zone_color

        stdscr.addstr(row, 0, VERTICAL, frame)
        stdscr.addstr(row, 1, content[:inner_width], color)
        try:
            stdscr.addstr(row, 1 + inner_width, VERTICAL, frame)
        except curses.error:
            pass
        row += 1

    # Footer separator
    stdscr.addstr(row, 0, sep[:width], frame)
    row += 1

    # Footer
    if no_weather:
        weather_age = None
    elif weather_time is None:
        weather_age = None
    else:
        weather_age = time.time() - weather_time
    footer_content = format_footer(weather_age, inner_width)
    stdscr.addstr(row, 0, VERTICAL, frame)
    stdscr.addstr(row, 1, footer_content[:inner_width], footer_color)
    try:
        stdscr.addstr(row, 1 + inner_width, VERTICAL, frame)
    except curses.error:
        pass
    row += 1

    # Bottom border
    bot = BOT_LEFT + HORIZONTAL * inner_width + BOT_RIGHT
    stdscr.addstr(row, 0, bot[:width], frame)

    stdscr.refresh()


def main_loop(stdscr, zones, refresh_minutes, no_weather):
    """Main curses event loop."""
    curses.curs_set(0)
    init_colors()
    stdscr.timeout(1000)  # 1 second timeout for key polling

    weather_data = {}
    weather_lock = threading.Lock()
    weather_time = None
    last_minute = None

    # Initial weather fetch
    if not no_weather:
        fetch_all_weather(zones, weather_data, weather_lock)
        weather_time = time.time()

    while True:
        now = time.time()

        # Auto-refresh weather
        if not no_weather and weather_time and (now - weather_time) >= refresh_minutes * 60:
            threading.Thread(
                target=_refresh_weather,
                args=(zones, weather_data, weather_lock),
                daemon=True,
            ).start()
            weather_time = now

        # Redraw on minute change or first draw
        current_minute = int(now) // 60
        if current_minute != last_minute:
            with weather_lock:
                draw_dashboard(stdscr, zones, dict(weather_data), weather_time, no_weather)
            last_minute = current_minute

        # Handle input
        try:
            key = stdscr.getch()
        except curses.error:
            key = -1

        if key == ord("q"):
            break
        elif key == ord("r") and not no_weather:
            threading.Thread(
                target=_refresh_weather,
                args=(zones, weather_data, weather_lock),
                daemon=True,
            ).start()
            weather_time = time.time()
            last_minute = None  # force redraw
        elif key == curses.KEY_RESIZE:
            last_minute = None  # force redraw on resize


def _refresh_weather(zones, weather_data, weather_lock):
    """Background weather refresh helper."""
    fetch_all_weather(zones, weather_data, weather_lock)


def main():
    args = parse_args()
    zones, refresh_minutes, no_weather = resolve_config(args)
    curses.wrapper(lambda stdscr: main_loop(stdscr, zones, refresh_minutes, no_weather))


if __name__ == "__main__":
    main()
```
